AIGC项目（1）

项目：吃了么GPT

(1)项目简介：最近经常做饭，想到可以训练一个专属的厨师模型来作为小助手，帮自己定菜谱，还能提供各类菜肴的做法，未来细分领域中专家GPT一定会成为趋势，主流应用也会往AIG的方向发展。

(3)项目代码：

import openai #调用OpenAI GPT-3 API的Python包。允许我们与OpenAI的自然语言处理模型进行交互
import streamlit as st #老熟人
# from streamlit_chat import message
from st_chat_message import message #显示聊天的样式
from PIL import Image # 用来加载和处理图片

proxy = {
'http': 'http://localhost:7890',
'https': 'http://localhost:7890'
}

openai.proxy = proxy;

# 这部分代码涉及到代理，用于与OpenAI进行通讯，用这个代理配置貌似无需科学上网

openai.api_key = 'YOUR API'
# 设置访问OpenAI的密钥
if 'prompts' not in st.session_state:
    st.session_state['prompts'] = [{"role": "system",
                                    # "content": "You are a helpful assistant. Answer as concisely as possible with a little humor expression."}
                                    "content": "You are a chef from China, "
                                               "please use all your talents to give me advice without using too many polite words."}
                                   ]
# 提示包含一个 "role" 和 "content"，用于在聊天界面中显示对话
if 'generated' not in st.session_state:
    st.session_state['generated'] = []
# 创建了名为 'generated' 的空列表，并将其存储在会话状态中。这个列表将用于存储由 GPT-3 生成的响应消息
if 'past' not in st.session_state:
    st.session_state['past'] = []
# 创建了名为 'past' 的空列表，并将其存储在会话状态中。这个列表将用于存储过去的聊天历史记录

def generate_response(prompt):
    st.session_state['prompts'].append({"role": "user", "content": prompt})
    # 添加用户输入的对话提示到会话状态中
    completion = openai.ChatCompletion.create(
        model="gpt-3.5-turbo",
        messages=st.session_state['prompts']
    )
    # 调用 OpenAI GPT-3 API 生成响应消息
    message = completion.choices[0].message.content
    # 提取生成的响应消息
    return message
    # 返回生成的响应消息

def end_click():
    st.session_state['prompts'] = [{"role": "system",
                                    # "content": "You are a helpful assistant. Answer as concisely as possible with a little humor expression."}
                                    "content": "You are a chef from China, "
                                               "please use all your talents to give me advice without using too many polite words."}
                                   ]
    # 重置对话提示为一个系统角色的初始提示
    st.session_state['past'] = []
    # 重置聊天历史记录
    st.session_state['generated'] = []
    # 重置生成的响应消息
    st.session_state['user'] = ""
    # 重置用户输入

def chat_click():
    if st.session_state['user'] != '':
        # 检查用户是否输入了内容
        chat_input = st.session_state['user']
        # 获取用户输入的对话内容
        output = generate_response(chat_input)
        # 生成对话的响应
        st.session_state['past'].append(chat_input)
        # 存储用户输入到聊天历史记录
        # store the output
        st.session_state['generated'].append(output)
        # 存储生成的响应到聊天历史记录
        st.session_state['prompts'].append({"role": "assistant", "content": output})
        # 将生成的响应添加到对话历史中，并标记为来自助手角色
        st.session_state['user'] = ""
        # 重置用户输入，以准备接受下一轮对话输入

    # 并将输入赋值给user_input变量。key参数用于在会话中保持这个组件的状态

    # "发送",

    end_button = st.button("新的聊天", on_click=end_click)
